004    package com.samaxes.maven.plugin.minify;
005    
006    import java.io.File;
007    import java.io.FileReader;
008    import java.io.FileWriter;
009    import java.io.IOException;
010    import java.io.Reader;
011    import java.io.Writer;
012    import java.util.List;
013    
014    import org.apache.maven.plugin.logging.Log;
015    
016    import com.yahoo.platform.yui.compressor.CssCompressor;
017    
018    /**
019     * Task for merging and compressing CSS files.
020     */
021    public class ProcessCSSFilesTask extends ProcessFilesTask {
022    
023        /**
024         * Task constructor.
025         * 
026         * @param log Maven plugin log
027         * @param bufferSize size of the buffer used to read source files.
028         * @param webappSourceDir web resources source directory
029         * @param webappTargetDir web resources target directory
030         * @param filesDir directory containing input files
031         * @param filenames filenames list
032         * @param finalFilename final filename
033         * @param linebreak split long lines after a specific column
034         */
035        public ProcessCSSFilesTask(Log log, Integer bufferSize, String webappSourceDir, String webappTargetDir,
036                String filesDir, List<String> filenames, String finalFilename, int linebreak) {
037            super(log, bufferSize, webappSourceDir, webappTargetDir, filesDir, filenames, finalFilename, linebreak);
038        }
039    
040        /**
041         * Minifies CSS file.
042         */
043        protected void minify() {
044            if (finalFile.exists()) {
045                String name = finalFile.getName();
046                String extension = name.substring(name.lastIndexOf('.'));
047                File destFile = new File(targetDir, name.replace(extension, suffix.concat(extension)));
048    
049                log.info("Minifying file " + name);
050                try {
051                    Reader reader = new FileReader(finalFile);
052                    Writer writer = new FileWriter(destFile);
053                    CssCompressor compressor = new CssCompressor(reader);
054    
055                    compressor.compress(writer, linebreak);
056                    reader.close();
057                    writer.close();
058                } catch (IOException e) {
059                    log.error(e.getMessage(), e);
060                }
061            }
062        }
063    }
