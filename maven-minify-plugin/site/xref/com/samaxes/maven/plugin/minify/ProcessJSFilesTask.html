47      public ProcessJSFilesTask(Log log, Integer bufferSize, String webappSourceDir, String webappTargetDir,
48              String filesDir, List<String> filenames, String finalFilename, int linebreak, boolean munge,
49              boolean verbose, boolean preserveAllSemiColons, boolean disableOptimizations) {
50          super(log, bufferSize, webappSourceDir, webappTargetDir, filesDir, filenames, finalFilename, linebreak);
51  
52          this.munge = munge;
53          this.verbose = verbose;
54          this.preserveAllSemiColons = preserveAllSemiColons;
55          this.disableOptimizations = disableOptimizations;
56      }
57  
58      /**
59       * Minifies JavaScript file.
60       */
61      protected void minify() {
62          if (finalFile.exists()) {
63              String name = finalFile.getName();
64              String extension = name.substring(name.lastIndexOf('.'));
65              File destFile = new File(targetDir, name.replace(extension, suffix.concat(extension)));
66  
67              log.info("Minifying file " + name);
68              try {
69                  Reader reader = new FileReader(finalFile);
70                  Writer writer = new FileWriter(destFile);
71                  JavaScriptCompressor compressor = new JavaScriptCompressor(reader, new JavaScriptErrorReporter(log,
72                          name));
73  
74                  compressor.compress(writer, linebreak, munge, verbose, preserveAllSemiColons, disableOptimizations);
75                  reader.close();
76                  writer.close();
77              } catch (IOException e) {
78                  log.error(e.getMessage(), e);
79              }
80          }
81      }
