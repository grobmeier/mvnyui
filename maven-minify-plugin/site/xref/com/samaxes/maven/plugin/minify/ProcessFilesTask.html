33  import java.util.List;

40  /**
41   * Task for merging and compressing a list of files.
42   */
43  public class ProcessFilesTask implements Runnable {
44  
45      private Log log;
46  
47      private Integer bufferSize;
48  
49      private File sourceDir;
50  
51      private File targetDir;
52  
53      private List<File> files = new ArrayList<File>();
54  
55      private File finalFile;
56  
57      private int linebreak;
58  
59      private boolean munge;
60  
61      private boolean verbose;
62  
63      private boolean preserveAllSemiColons;
64  
65      private boolean disableOptimizations;
66  
67      private final String suffix = ".min";
68  
69      /**
70       * Task constructor.
71       * 
72       * @param log Maven plugin log
73       * @param bufferSize size of the buffer used to read source files.
74       * @param webappSourceDir web resources source directory
75       * @param webappTargetDir web resources target directory
76       * @param filesDir directory containing input files
77       * @param filenames filenames list
78       * @param finalFilename final filename
79       * @param linebreak split long lines after a specific column
80       * @param munge minify only
81       * @param verbose display informational messages and warnings
82       * @param preserveAllSemiColons preserve unnecessary semicolons
83       * @param disableOptimizations disable all the built-in micro optimizations
84       */
85      public ProcessFilesTask(Log log, Integer bufferSize, String webappSourceDir, String webappTargetDir,
86              String filesDir, List<String> filenames, String finalFilename, int linebreak, boolean munge,
87              boolean verbose, boolean preserveAllSemiColons, boolean disableOptimizations) {
88          this.log = log;
89          this.bufferSize = bufferSize;
90          this.sourceDir = new File(webappSourceDir.concat(File.separator).concat(filesDir));
91          this.targetDir = new File(webappTargetDir.concat(File.separator).concat(filesDir));
92          for (String filename : filenames) {
93              files.add(new File(sourceDir, filename));
94          }
95          if (targetDir.exists() || targetDir.mkdirs()) {
96              this.finalFile = new File(targetDir, finalFilename);
97          }
98  
99          this.linebreak = linebreak;
100         this.munge = munge;
101         this.verbose = verbose;
102         this.preserveAllSemiColons = preserveAllSemiColons;
103         this.disableOptimizations = disableOptimizations;
104     }
105 
106     /**
107      * Method executed by the thread.
108      */
109     public void run() {
110         mergeFiles();
111         minify(finalFile);
112     }
113 
114     /**
115      * Merges a list of files.
116      */
117     private void mergeFiles() {
118         ListOfFiles listOfFiles = new ListOfFiles(this.log, files);
119         log.info("Merging files " + listOfFiles.toString());
120 
121         if (listOfFiles.size() > 0) {
122             try {
123                 SequenceInputStream sequence = new SequenceInputStream(listOfFiles);
124                 OutputStream out = new FileOutputStream(finalFile);
125                 byte[] buffer = new byte[bufferSize];
126                 int length;
127                 while ((length = sequence.read(buffer)) > 0) {
128                     out.write(buffer, 0, length);
129                 }
130 
131                 sequence.close();
132                 out.close();
133             } catch (IOException e) {
134                 log.error("An error has occurred while concatenating files.", e);
135             }
136         }
137     }
138 
139     /**
140      * Minifies a CSS or JavaScript file.
141      * 
142      * @param sourceFile the source file
143      */
144     private void minify(File sourceFile) {
145         log.info("Minifying file " + sourceFile.getName());
146 
147         if (sourceFile.exists()) {
148             String extension = sourceFile.getName().substring(sourceFile.getName().lastIndexOf('.'));
149             File destFile = new File(targetDir, sourceFile.getName().replace(extension, suffix.concat(extension)));
150 
151             try {
152                 Reader reader = new FileReader(sourceFile);
153                 Writer writer = new FileWriter(destFile);
154 
155                 if (".css".equalsIgnoreCase(extension)) {
156                     CssCompressor compressor = new CssCompressor(reader);
157                     compressor.compress(writer, linebreak);
158                 } else if (".js".equalsIgnoreCase(extension)) {
159                     JavaScriptCompressor compressor = new JavaScriptCompressor(reader, new JavaScriptErrorReporter(log));
160                     compressor.compress(writer, linebreak, munge, verbose, preserveAllSemiColons, disableOptimizations);
161                 }
162 
163                 reader.close();
164                 writer.close();
165             } catch (IOException e) {
166                 log.error(e.getMessage(), e);
167             }
168         }
169     }
