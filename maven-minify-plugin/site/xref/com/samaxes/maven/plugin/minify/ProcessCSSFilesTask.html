44      /**
45       * Minifies CSS file.
46       */
47      protected void minify() {
48          if (finalFile.exists()) {
49              String name = finalFile.getName();
50              String extension = name.substring(name.lastIndexOf('.'));
51              File destFile = new File(targetDir, name.replace(extension, SUFFIX.concat(extension)));
52  
53              log.info("Minifying file [" + name + "]");
54              try {
55                  Reader reader = new FileReader(finalFile);
56                  Writer writer = new FileWriter(destFile);
57                  CssCompressor compressor = new CssCompressor(reader);
58  
59                  compressor.compress(writer, linebreak);
60                  reader.close();
61                  writer.close();
62              } catch (IOException e) {
63                  log.error(e.getMessage(), e);
64              }
65          }
66      }
