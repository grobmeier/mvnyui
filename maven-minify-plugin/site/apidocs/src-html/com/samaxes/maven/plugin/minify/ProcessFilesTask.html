021    package com.samaxes.maven.plugin.minify;
022    
023    import java.io.File;
024    import java.io.FileOutputStream;
025    import java.io.IOException;
026    import java.io.OutputStream;
027    import java.io.SequenceInputStream;
028    import java.util.ArrayList;
029    import java.util.List;
030    
031    import org.apache.maven.plugin.logging.Log;
032    import org.codehaus.plexus.util.DirectoryScanner;
033    import org.codehaus.plexus.util.StringUtils;
034    
035    import com.samaxes.maven.plugin.common.ListOfFiles;
036    
037    /**
038     * Abstract class for merging and compressing a files list.
039     */
040    public abstract class ProcessFilesTask implements Runnable {
041    
042        protected static final String SUFFIX = ".min";
043    
044        private static final String[] EMPTY_STRING_ARRAY = {};
045    
046        protected Log log;
047    
048        protected File targetDir;
049    
050        protected File finalFile;
051    
052        protected int linebreak;
053    
054        private Integer bufferSize;
055    
056        private File sourceDir;
057    
058        private List<File> files = new ArrayList<File>();
059    
060        /**
061         * Task constructor.
062         * 
063         * @param log Maven plugin log
064         * @param bufferSize size of the buffer used to read source files.
065         * @param webappSourceDir web resources source directory
066         * @param webappTargetDir web resources target directory
067         * @param filesDir directory containing input files
068         * @param filenames filenames list
069         * @param sourceIncludes comma separated list of source files to include
070         * @param sourceExcludes comma separated list of source files to exclude
071         * @param finalFilename final filename
072         * @param linebreak split long lines after a specific column
073         */
074        public ProcessFilesTask(Log log, Integer bufferSize, String webappSourceDir, String webappTargetDir,
075                String filesDir, List<String> filenames, String sourceIncludes, String sourceExcludes,
076                String finalFilename, int linebreak) {
077            this.log = log;
078            this.bufferSize = bufferSize;
079            this.linebreak = linebreak;
080            this.sourceDir = new File(webappSourceDir.concat(File.separator).concat(filesDir));
081            this.targetDir = new File(webappTargetDir.concat(File.separator).concat(filesDir));
082    
083            List<File> includedFiles = (sourceIncludes != null && !"".equals(sourceIncludes)) ? getFilesToInclude(
084                    sourceDir, sourceIncludes, sourceExcludes) : new ArrayList<File>();
085            for (String filename : filenames) {
086                File sourceFile = new File(sourceDir, filename);
087                log.debug("Adding source file [" + sourceFile.getName() + "]");
088                files.add(sourceFile);
089            }
090            for (File includedFile : includedFiles) {
091                if (!files.contains(includedFile)) {
092                    log.debug("Adding source file [" + includedFile.getName() + "]");
093                    files.add(includedFile);
094                }
095            }
096    
097            if (targetDir.exists() || targetDir.mkdirs()) {
098                this.finalFile = new File(targetDir, finalFilename);
099            }
100        }
101    
102        /**
103         * Method executed by the thread.
104         */
105        public void run() {
106            mergeFiles();
107            minify();
108        }
109    
110        /**
111         * Returns a string array of the includes to be used when adding source files.
112         * 
113         * @param include comma separated list of source files to include
114         * @return an array of tokens to include
115         */
116        private String[] getIncludes(String include) {
117            return StringUtils.split(StringUtils.defaultString(include), ",");
118        }
119    
120        /**
121         * Returns a string array of the excludes to be used when adding source files.
122         * 
123         * @param exclude comma separated list of source files to exclude
124         * @return an array of tokens to exclude
125         */
126        private String[] getExcludes(String exclude) {
127            return (StringUtils.isNotEmpty(exclude)) ? StringUtils.split(exclude, ",") : EMPTY_STRING_ARRAY;
128        }
129    
130        /**
131         * Returns the files to copy. Even if the excludes are <code>null</code>, the default excludes are used.
132         * 
133         * @param baseDir the base directory to start from
134         * @param include comma separated list of source files to include
135         * @param exclude comma separated list of source files to exclude
136         * @return the files to copy
137         */
138        private List<File> getFilesToInclude(File baseDir, String include, String exclude) {
139            DirectoryScanner scanner = new DirectoryScanner();
140            List<File> includedFiles = new ArrayList<File>();
141            String[] includedFilenames;
142            String[] includes = getIncludes(include);
143            String[] excludes = getExcludes(exclude);
144    
145            scanner.setBasedir(baseDir);
146    
147            if (excludes != null) {
148                scanner.setExcludes(excludes);
149            }
150            scanner.addDefaultExcludes();
151    
152            if (includes != null && includes.length > 0) {
153                scanner.setIncludes(includes);
154            }
155    
156            scanner.scan();
157            includedFilenames = scanner.getIncludedFiles();
158    
159            for (String includedFilename : includedFilenames) {
160                includedFiles.add(new File(baseDir, includedFilename));
161            }
162    
163            return includedFiles;
164        }
165    
166        /**
167         * Merges files list.
168         */
169        private void mergeFiles() {
170            ListOfFiles listOfFiles = new ListOfFiles(this.log, files);
171    
172            log.info("Merging files " + listOfFiles.toString());
173            if (listOfFiles.size() > 0) {
174                try {
175                    SequenceInputStream sequence = new SequenceInputStream(listOfFiles);
176                    OutputStream out = new FileOutputStream(finalFile);
177                    byte[] buffer = new byte[bufferSize];
178                    int length;
179                    while ((length = sequence.read(buffer)) > 0) {
180                        out.write(buffer, 0, length);
181                    }
182    
183                    sequence.close();
184                    out.close();
185                } catch (IOException e) {
186                    log.error("An error has occurred while concatenating files.", e);
187                }
188            }
189        }
190    
191        /**
192         * Minifies source file.
193         */
194        abstract void minify();
195    }
