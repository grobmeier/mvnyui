004    package com.samaxes.maven.plugin.minify;
005    
006    import java.io.File;
007    import java.io.FileReader;
008    import java.io.FileWriter;
009    import java.io.IOException;
010    import java.io.Reader;
011    import java.io.Writer;
012    import java.util.List;
013    
014    import org.apache.maven.plugin.logging.Log;
015    
016    import com.samaxes.maven.plugin.common.JavaScriptErrorReporter;
017    import com.yahoo.platform.yui.compressor.JavaScriptCompressor;
018    
019    /**
020     * Task for merging and compressing JavaScript files.
021     */
022    public class ProcessJSFilesTask extends ProcessFilesTask {
023    
024        private boolean munge;
025    
026        private boolean verbose;
027    
028        private boolean preserveAllSemiColons;
029    
030        private boolean disableOptimizations;
031    
032        /**
033         * Task constructor.
034         * 
035         * @param log Maven plugin log
036         * @param bufferSize size of the buffer used to read source files.
037         * @param webappSourceDir web resources source directory
038         * @param webappTargetDir web resources target directory
039         * @param filesDir directory containing input files
040         * @param filenames filenames list
041         * @param sourceIncludes comma separated list of source files to include
042         * @param sourceExcludes comma separated list of source files to exclude
043         * @param finalFilename final filename
044         * @param linebreak split long lines after a specific column
045         * @param munge minify only
046         * @param verbose display informational messages and warnings
047         * @param preserveAllSemiColons preserve unnecessary semicolons
048         * @param disableOptimizations disable all the built-in micro optimizations
049         */
050        public ProcessJSFilesTask(Log log, Integer bufferSize, String webappSourceDir, String webappTargetDir,
051                String filesDir, List<String> filenames, String sourceIncludes, String sourceExcludes,
052                String finalFilename, int linebreak, boolean munge, boolean verbose, boolean preserveAllSemiColons,
053                boolean disableOptimizations) {
054            super(log, bufferSize, webappSourceDir, webappTargetDir, filesDir, filenames, sourceIncludes, sourceExcludes,
055                    finalFilename, linebreak);
056    
057            this.munge = munge;
058            this.verbose = verbose;
059            this.preserveAllSemiColons = preserveAllSemiColons;
060            this.disableOptimizations = disableOptimizations;
061        }
062    
063        /**
064         * Minifies JavaScript file.
065         */
066        protected void minify() {
067            if (finalFile.exists()) {
068                String name = finalFile.getName();
069                String extension = name.substring(name.lastIndexOf('.'));
070                File destFile = new File(targetDir, name.replace(extension, SUFFIX.concat(extension)));
071    
072                log.info("Minifying file [" + name + "]");
073                try {
074                    Reader reader = new FileReader(finalFile);
075                    Writer writer = new FileWriter(destFile);
076                    JavaScriptCompressor compressor = new JavaScriptCompressor(reader, new JavaScriptErrorReporter(log,
077                            name));
078    
079                    compressor.compress(writer, linebreak, munge, verbose, preserveAllSemiColons, disableOptimizations);
080                    reader.close();
081                    writer.close();
082                } catch (IOException e) {
083                    log.error(e.getMessage(), e);
084                }
085            }
086        }
087    }
