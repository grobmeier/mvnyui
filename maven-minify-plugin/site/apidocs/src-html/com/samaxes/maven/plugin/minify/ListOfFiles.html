021    package com.samaxes.maven.plugin.minify;
022    
023    import java.io.File;
024    import java.io.FileInputStream;
025    import java.io.FileNotFoundException;
026    import java.io.InputStream;
027    import java.util.Enumeration;
028    import java.util.List;
029    import java.util.NoSuchElementException;
030    
031    import org.apache.maven.plugin.logging.Log;
032    
033    /**
034     * {@code ListOfFiles} is used to initialize the SequenceInputStream which uses {@code ListOfFiles} to get a new
035     * InputStream for every file listed.
036     */
037    public class ListOfFiles implements Enumeration<InputStream> {
038    
039        private Log log;
040    
041        private List<File> files;
042    
043        private int current = 0;
044    
045        /**
046         * ListOfFiles public constructor.
047         * 
048         * @param log Maven plugin log
049         * @param files list of files
050         */
051        public ListOfFiles(Log log, List<File> files) {
052            this.log = log;
053            this.files = files;
054        }
055    
056        /**
057         * Tests if this enumeration contains more elements.
058         * 
059         * @return <code>true</code> if and only if this enumeration object contains at least one more element to provide;
060         *         <code>false</code> otherwise.
061         */
062        public boolean hasMoreElements() {
063            return (current < files.size()) ? true : false;
064        }
065    
066        /**
067         * Returns the next element of this enumeration if this enumeration object has at least one more element to provide.
068         * 
069         * @return the next element of this enumeration.
070         * @exception NoSuchElementException if no more elements exist.
071         */
072        public InputStream nextElement() {
073            InputStream is = null;
074    
075            if (!hasMoreElements())
076                throw new NoSuchElementException("No more files.");
077            else {
078                try {
079                    File nextElement = files.get(current);
080                    current++;
081                    is = new FileInputStream(nextElement);
082                } catch (FileNotFoundException e) {
083                    log.error("The file [" + files.get(current).getName() + "] was not found.", e);
084                }
085            }
086    
087            return is;
088        }
089    
090        /**
091         * Returns the list of files contained in this object.
092         * 
093         * @return the list of files
094         */
095        public String toString() {
096            StringBuilder stringBuilder = new StringBuilder("[");
097            for (File file : files) {
098                if (stringBuilder.length() > 1) {
099                    stringBuilder.append(", ");
100                }
101                stringBuilder.append(file.getName());
102            }
103    
104            return stringBuilder.append("]").toString();
105        }
106    
107        /**
108         * Returns the number of elements in this list.
109         * 
110         * @return the number of elements in this list
111         */
112        public int size() {
113            return files.size();
114        }
115    }
