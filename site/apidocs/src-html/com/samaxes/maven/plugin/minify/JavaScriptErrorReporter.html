021    package com.samaxes.maven.plugin.minify;
022    
023    import org.apache.maven.plugin.logging.Log;
024    import org.mozilla.javascript.ErrorReporter;
025    import org.mozilla.javascript.EvaluatorException;
026    
027    /**
028     * Reports any error occurring during JavaScript files compression.
029     */
030    public class JavaScriptErrorReporter implements ErrorReporter {
031    
032        private Log log;
033    
034        /**
035         * Error reporter constructor.
036         * 
037         * @param log Maven plugin log
038         */
039        public JavaScriptErrorReporter(Log log) {
040            this.log = log;
041        }
042    
043        /**
044         * Reports a warning.
045         * 
046         * @param message a String describing the warning
047         * @param sourceName a String describing the JavaScript source where the warning occured; typically a filename or
048         *        URL
049         * @param line the line number associated with the warning
050         * @param lineSource the text of the line (may be null)
051         * @param lineOffset the offset into lineSource where problem was detected
052         */
053        public void warning(String message, String sourceName, int line, String lineSource, int lineOffset) {
054            if (line < 0) {
055                log.warn(message);
056            } else {
057                log.warn(line + ':' + lineOffset + ':' + message);
058            }
059        }
060    
061        /**
062         * Reports an error. If execution has not yet begun, the JavaScript engine is free to find additional errors rather
063         * than terminating the translation. It will not execute a script that had errors, however.
064         * 
065         * @param message a String describing the warning
066         * @param sourceName a String describing the JavaScript source where the warning occured; typically a filename or
067         *        URL
068         * @param line the line number associated with the warning
069         * @param lineSource the text of the line (may be null)
070         * @param lineOffset the offset into lineSource where problem was detected
071         */
072        public void error(String message, String sourceName, int line, String lineSource, int lineOffset) {
073            if (line < 0) {
074                log.error(message);
075            } else {
076                log.error(line + ':' + lineOffset + ':' + message);
077            }
078        }
079    
080        /**
081         * Creates an EvaluatorException that may be thrown. runtimeErrors, unlike errors, will always terminate the current
082         * script.
083         * 
084         * @param message a String describing the warning
085         * @param sourceName a String describing the JavaScript source where the warning occured; typically a filename or
086         *        URL
087         * @param line the line number associated with the warning
088         * @param lineSource the text of the line (may be null)
089         * @param lineOffset the offset into lineSource where problem was detected
090         */
091        public EvaluatorException runtimeError(String message, String sourceName, int line, String lineSource,
092                int lineOffset) {
093            error(message, sourceName, line, lineSource, lineOffset);
094    
095            return new EvaluatorException(message);
096        }
097    }
